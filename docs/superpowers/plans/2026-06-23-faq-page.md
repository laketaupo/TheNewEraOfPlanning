# FAQ Page Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `/faq` page grouped by pillar, with cross-references to topics and glossary terms, a client-side text filter, and a question-mark icon in the top-right nav bar.

**Architecture:** Mirrors the existing glossary pattern exactly — a flat JSON content file (`faq.json`), a lib module (`faq.ts`) that loads and validates at build time, and an Astro page (`faq.astro`). Two existing files are modified to add navigation links.

**Tech Stack:** Astro 4, Tailwind CSS 3, TypeScript, static site generation (no runtime server).

## Global Constraints

- No test runner or linter is configured — verification is `npm run build` (must succeed with zero errors) followed by manual visual check with `npm run dev`.
- `npm run build` outputs to `.vercel/output/static/` via `@astrojs/vercel/static` adapter.
- Tailwind classes must appear as complete strings in source — no string interpolation.
- Pillar values must be exactly `technology`, `process`, `data`, `people`.
- `seeAlso` refs must be valid `chapter-slug/topic-slug` pairs from `src/content/chapters/`. Leave as `[]` if unsure — the build will throw on bad refs.
- `related` refs must be valid glossary slugs from `src/content/glossary.json`. Leave as `[]` if unsure.

- [ ] **Step 2: Create `src/lib/faq.ts`**

```typescript
import faqRaw from '../content/faq.json';
import { getTopics } from './chapters';
import { getGlossaryTerms } from './glossary';

export interface FaqEntry {
  slug: string;
  question: string;
  answer: string;
  pillar: string;
  seeAlso: string[];
  related: string[];
}

const VALID_PILLARS = new Set(['technology', 'process', 'data', 'people']);

export function getFaqEntries(): FaqEntry[] {
  return (faqRaw as FaqEntry[]).map(e => ({
    ...e,
    seeAlso: e.seeAlso ?? [],
    related: e.related ?? [],
  }));
}

export function validateFaq(): void {
  const entries = getFaqEntries();
  const topics = getTopics();
  const topicIds = new Set(topics.map(t => `${t.chapterSlug}/${t.slug}`));
  const glossaryTerms = getGlossaryTerms();
  const termSlugs = new Set(glossaryTerms.map(t => t.slug));

  for (const entry of entries) {
    if (!VALID_PILLARS.has(entry.pillar)) {
      throw new Error(
        `FAQ "${entry.slug}": invalid pillar "${entry.pillar}". ` +
        `Must be one of: technology, process, data, people.`
      );
    }
    for (const ref of entry.seeAlso) {
      if (!topicIds.has(ref)) {
        throw new Error(
          `FAQ "${entry.slug}": invalid seeAlso reference "${ref}". ` +
          `Check that the chapter-slug/topic-slug exists in src/content/chapters/.`
        );
      }
    }
    for (const ref of entry.related) {
      if (!termSlugs.has(ref)) {
        throw new Error(
          `FAQ "${entry.slug}": invalid related reference "${ref}". ` +
          `No glossary term with that slug exists.`
        );
      }
    }
  }
}
```

```astro
import BaseLayout from '../layouts/BaseLayout.astro';
import { getFaqEntries, validateFaq } from '../lib/faq';
import { getTopics } from '../lib/chapters';
import { getGlossaryTerms } from '../lib/glossary';

validateFaq();

const entries = getFaqEntries();
const topics = getTopics();
const glossaryTerms = getGlossaryTerms();
const topicMap = Object.fromEntries(topics.map(t => [`${t.chapterSlug}/${t.slug}`, t]));
const glossaryMap = Object.fromEntries(glossaryTerms.map(t => [t.slug, t]));

const PILLAR_ORDER = ['technology', 'process', 'data', 'people'] as const;
const PILLAR_LABELS: Record<string, string> = {
  technology: 'Technology',
  process: 'Process',
  data: 'Data',
  people: 'People',
};

const pillarsWithEntries = PILLAR_ORDER.filter(p => entries.some(e => e.pillar === p));

<BaseLayout title="FAQ — How Planning Software Works" description="Frequently asked questions about planning software, S&OP, data, and people in the planning process.">
  <header class="fixed top-0 left-0 right-0 z-50 flex items-center justify-between px-6 py-3">
    <div class="flex items-center gap-3">
      <a href="/" title="Home" class="flex h-5 items-center text-gray-500 dark:text-neutral-400 hover:text-gray-900 dark:hover:text-white transition-colors">
        <svg xmlns="http://www.w3.org/2000/svg" class="w-4 h-4" fill="none" viewBox="0 0 24 24" stroke="currentColor">

        </svg>
      </a>
    </div>
  </header>

  <main class="max-w-3xl mx-auto px-6 pt-24 pb-32">
    <h1 class="text-3xl font-bold text-gray-900 dark:text-white mb-2">Frequently Asked Questions</h1>
    <p class="text-gray-500 dark:text-neutral-400 mb-6">{entries.length} questions across {pillarsWithEntries.length} pillars.</p>

    <!-- Text filter -->
    <div class="mb-8">
      <input
        id="faq-filter"
        type="search"
        placeholder="Filter questions…"
        class="w-full rounded-lg border border-gray-200 dark:border-neutral-700 bg-white dark:bg-neutral-900 px-4 py-2 text-sm text-gray-900 dark:text-white placeholder-gray-400 dark:placeholder-neutral-500 focus:outline-none focus:ring-2 focus:ring-indigo-500 dark:focus:ring-indigo-400"
      />
    </div>

    <!-- Pillar jump nav -->
    <nav class="flex flex-wrap gap-1.5 mb-10" aria-label="Jump to pillar">
      {pillarsWithEntries.map(pillar => (
        <a
          href={`#pillar-${pillar}`}
          class="px-3 h-7 flex items-center justify-center rounded text-xs font-semibold text-gray-500 dark:text-neutral-400 hover:bg-indigo-50 dark:hover:bg-indigo-500/10 hover:text-indigo-600 dark:hover:text-indigo-400 transition-colors"
        >
          {PILLAR_LABELS[pillar]}
        </a>
      ))}
    </nav>

    <!-- Empty state -->
    <p id="faq-empty" class="hidden text-sm text-gray-400 dark:text-neutral-500 text-center py-10">No questions matched.</p>

    <!-- Entries grouped by pillar -->
    {PILLAR_ORDER.map(pillar => {
      const group = entries.filter(e => e.pillar === pillar);
      if (group.length === 0) return null;
      return (
        <section id={`pillar-${pillar}`} class="mb-10 faq-section">
          <h2 class="text-xs font-bold uppercase tracking-widest text-gray-400 dark:text-neutral-600 mb-4 border-b border-gray-100 dark:border-neutral-800 pb-2">
            {PILLAR_LABELS[pillar]}
          </h2>
          <dl class="space-y-6">
            {group.map(entry => {
              const seeAlsoTopics = entry.seeAlso.map(ref => topicMap[ref]).filter(Boolean);
              const relatedTerms = entry.related.map(slug => glossaryMap[slug]).filter(Boolean);
              return (
                <div id={entry.slug} class="scroll-mt-20 faq-entry" data-search={`${entry.question} ${entry.answer}`.toLowerCase()}>
                  <dt class="text-base font-semibold text-gray-900 dark:text-white mb-1">{entry.question}</dt>
                  <dd class="text-sm text-gray-600 dark:text-neutral-400 leading-relaxed mb-2">{entry.answer}</dd>
                  {seeAlsoTopics.length > 0 && (
                    <p class="text-xs text-gray-400 dark:text-neutral-600">
                      See:{' '}
                      {seeAlsoTopics.map((t, i) => (
                        <Fragment>
                          <a href={t.url} class="text-indigo-600 dark:text-indigo-400 hover:underline">{t.title}</a>
                          {i < seeAlsoTopics.length - 1 ? ', ' : ''}
                        </Fragment>
                      ))}
                    </p>
                  )}
                  {relatedTerms.length > 0 && (
                    <p class="text-xs text-gray-400 dark:text-neutral-600 mt-0.5">
                      Related:{' '}
                      {relatedTerms.map((t, i) => (
                        <Fragment>
                          <a href={`/glossary#${t.slug}`} class="text-indigo-600 dark:text-indigo-400 hover:underline">{t.term}</a>
                          {i < relatedTerms.length - 1 ? ', ' : ''}
                        </Fragment>
                      ))}
                    </p>
                  )}
                </div>
              );
            })}
          </dl>
        </section>
      );
    })}
  </main>

  <script>
    const input = document.getElementById('faq-filter') as HTMLInputElement;
    const empty = document.getElementById('faq-empty')!;
    const sections = document.querySelectorAll<HTMLElement>('.faq-section');

    input.addEventListener('input', () => {
      const query = input.value.toLowerCase().trim();
      let totalVisible = 0;

      sections.forEach(section => {
        const entries = section.querySelectorAll<HTMLElement>('.faq-entry');
        let sectionVisible = 0;

        entries.forEach(entry => {
          const text = entry.dataset.search ?? '';
          const matches = !query || text.includes(query);
          entry.style.display = matches ? '' : 'none';
          if (matches) sectionVisible++;
        });

        section.style.display = sectionVisible > 0 ? '' : 'none';
        totalVisible += sectionVisible;
      });

      empty.classList.toggle('hidden', totalVisible > 0);
    });
  </script>
</BaseLayout>
```
